 * Permite formatear un número con los espacios necesarios delante.
 * Modificación de http://stackoverflow.com/a/10073788
 */
function pad (n, width, z)
{
	z = z || ' ';
	n = n + '';
	return n.length >= width ? n : new Array(width - n.length + 1).join(z) + n;
}
